(RECEITA-LIST)
SELECT
REC.ID, 
REC.NOME, 
REC.CATEGORIA,
COUNT(RECING.RECEITA_ID) AS QTD_ING

FROM 
RECEITA REC, 
RECEITA_INGREDIENTES RECING 

WHERE REC.ID = RECING.RECEITA_ID

GROUP BY REC.ID
