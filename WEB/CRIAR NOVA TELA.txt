(RECEITA-LIST)
SELECT DISTINCT 
R.ID,
R.NOME, 
R.CATEGORIA, 

(SELECT COUNT(Y.ID)
FROM RECEITA X,
RECEITA_INGREDIENTES Y
WHERE X.ID = Y.RECEITA_ID
) AS QTD_INGREDIENTES,

--VAI DEIXAR LENTO(?)
CASE
    WHEN 
	 	(SELECT COUNT(Z.ID)
		FROM RECEITA W,
		MODOPREPARO Z
		WHERE W.ID = Z.RECEITA_ID
		) <= 5 THEN "FÁCIL"
    WHEN 
	 	(SELECT COUNT(Z.ID)
		FROM RECEITA W,
		MODOPREPARO Z
		WHERE W.ID = Z.RECEITA_ID
		) >= 10 THEN "MÉDIO"
    ELSE "DIFÍCIL"
END AS NIVEL

FROM RECEITA R
